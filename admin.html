📈 إحصائيات الزوار — Google Analytics

1
سجّل على analytics.google.com
2
أنشئ Property جديدة للموقع ssamau.com
3
احصل على Measurement ID (مثال: G-XXXXXXXXXX) واستبدل G-XXXXXXXXXX في ملف index.html
4
ستظهر إحصائيات الزوار (عددهم، موقعهم، الجهاز) في لوحة Google Analytics خلال 24 ساعة
💡 تتبع مجاني كامل: عدد الزوار، البلد، المدينة، الجهاز، الصفحات الأكثر زيارة، مدة البقاء

🔗 ربط Google Sheets (اختياري)

1
أنشئ Google Sheet بالتبويبات: Events Team Members Hours Users News Partners
2
في Extensions → Apps Script الصق الكود أدناه وانشره كـ Web App
3
الصق رابط Web App في الحقل أدناه
const ID='YOUR_SHEET_ID'; function doGet(){const ss=SpreadsheetApp.openById(ID);const t=['Events','Team','Members','Hours','Users','News','Partners']; const o={};t.forEach(n=>{const s=ss.getSheetByName(n);if(s)o[n]=s.getDataRange().getValues();});return R(o);} function doPost(e){const b=JSON.parse(e.postData.contents);const s=SpreadsheetApp.openById(ID).getSheetByName(b.sheet); if(b.action=='add')s.appendRow(b.row);else if(b.action=='update')s.getRange(b.ri+1,1,1,b.row.length).setValues([b.row]);else if(b.action=='delete')s.deleteRow(b.ri+1);return R({ok:true});} const R=o=>ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON);
